/*
 *Write a C program to convert specified days into years, weeks and days.
 *Note: Ignore leap year.
 *Test Data :
 * Number of days : 1329
 * Expected Output :
 * Years: 3
 * Weeks: 33
 * Days: 3
*/
#include <stdio.h>
int main()
{
	int days, years, weeks;
	printf ("Enter the Days:");
	scanf ("%d",&days);	
    years = days / 365;	
    weeks = (days % 365)/7;
    days = (days % 365)%7;
    printf("Years: %d\n", years);
    printf("Weeks: %d\n", weeks);
    printf("Days: %d \n", days);
    return 0;
}

/*
 *Write a C program that multiplies a given number by 18 
 *without using multiplication operator and without adding the number 18 times
*/
#include <stdio.h>
int main()
{
	int result, num;	
	printf ("Enter the number:");
	scanf ("%d",&num);
	result = (num<<4)+(num<<1) ;
	printf("number * 18: %d\n", result);
	return 0;
}


/*

 *without using any arithmetic or relational operators
*/

{

/*
 *Write a C program to check whether a character is an alphabet, digit or special character. 
 *Test Data : @
 * Expected Output : This is a special character.
*/
#include<stdio.h>

int main()
{
	char x;
	printf("Enter a character: ");
	scanf("%c",&x);
	if(x >= '0' && x <= '9')
	{
		printf("This is a digit");
	}
	else if((x >= 'A' && x <= 'Z')||(x >= 'a' && x <= 'z'))
	{
		printf("This is an alphabet");
	}
	else
	{
		printf("This is a special character");
	}
	
	
	return 0;
}

/*
 *Write a C program that accepts a positive integer less than 500 and prints out the sum of the digits of this number. 
 *Input a positive number less than 500:
 * Sum of the digits of 347 is 14
*/
#include<stdio.h>
int main()
{
    int x,sum = 0;
	int temp;
	printf("Enter a number : ");
	scanf("%d",&x);
	temp=x;
	if(x < 500 && x >= 0)
	{
		while(x != 0)
		{
			sum += x%10;
			x /= 10;
		}
		printf("Sum of the digits of %d is %d\n",temp,sum);
	}
	else
	{
		printf("Invalid number");
	}
	
	return 0;
}
